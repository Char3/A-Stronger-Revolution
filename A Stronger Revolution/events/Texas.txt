country_event = {
	id = 73848
	title = "Texan Settlers"
	desc = "TEX1"
	picture = "Administration"
	trigger = {
		NOT = { has_country_flag = texans_blocked }
		NOT = { tag = USA }
		NOT = { tag = CSA }
		NOT = { tag = TEX }
		any_owned_province = {
			is_core = TXS
		}
		132 = { owned_by = TEX }
		NOT = { year = 1852 }
	}
	
	mean_time_to_happen = {
		years = 1
	}
	
	
	option = {
		name = "Concerning but welcome."
		any_owned = { 
			any_pop = {
				limit = { culture = texan }
				reduce_pop = 1.18
			}
		}
		any_owned = {
			any_pop = {
				limit = { culture = mexican }
				militancy = 0.2
			}
		}
	}
}
